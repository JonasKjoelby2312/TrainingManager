
GET_ALL_EMPLOYEES
(
SELECT 
    e.employee_id AS EmployeeId,
    e.employee_initials AS Initials,
    e.employee_email AS Email,
    e.employee_is_active AS IsActive,
    STRING_AGG(tr.role_name, ', ') AS Roles  -- Aggregates roles per employee
FROM employees e
LEFT JOIN employee_roles er ON er.fk_employee_id = e.employee_id
LEFT JOIN treat_roles tr ON tr.role_id = er.fk_role_id
GROUP BY 
    e.employee_id, e.employee_initials, e.employee_email, e.employee_is_active
ORDER BY e.employee_id;
)

GET_TRAINING_STATUS
(
WITH selected_employee AS (
    SELECT employee_id, employee_initials
    FROM employees
    WHERE employee_initials = 'mm'
),
active_procedures AS (
    SELECT tp.procedure_id, tp.procedure_name
    FROM treat_procedures tp
    JOIN revisions r ON tp.fk_revision_id = r.revision_id
    WHERE r.revision_is_active = 1
),
employee_roles_expanded AS (
    SELECT er.fk_role_id
    FROM selected_employee se
    JOIN employee_roles er ON se.employee_id = er.fk_employee_id
),
training_status_base AS (
    SELECT 
        ap.procedure_name,
        MAX(
            CASE 
                WHEN t.ticket_id IS NOT NULL THEN 3
                WHEN rtt.required_type = 1 THEN 2
                WHEN rtt.required_type = 2 THEN 1
                ELSE 0
            END
        ) AS StatusRank
    FROM active_procedures ap
    CROSS JOIN selected_employee se
    LEFT JOIN employee_roles er ON er.fk_employee_id = se.employee_id
    LEFT JOIN required_training_types rtt 
        ON rtt.fk_role_id = er.fk_role_id 
        AND rtt.fk_treat_procedure_id = ap.procedure_id
    LEFT JOIN tickets t 
        ON t.fk_employee_id = se.employee_id 
        AND t.fk_treat_procedure_id = ap.procedure_id
    GROUP BY ap.procedure_name
)
SELECT 
    procedure_name,
    CASE StatusRank
        WHEN 3 THEN 'Training Completed'
        WHEN 2 THEN 'Training Missing'
        WHEN 1 THEN 'Training Optional'
        ELSE 'Place holder'
    END AS TrainingStatus
FROM training_status_base
ORDER BY procedure_name;WITH selected_employee AS (
    SELECT employee_id, employee_initials
    FROM employees
    WHERE employee_initials = 'mm'
),
active_procedures AS (
    SELECT tp.procedure_id, tp.procedure_name
    FROM treat_procedures tp
    JOIN revisions r ON tp.fk_revision_id = r.revision_id
    WHERE r.revision_is_active = 1
),
employee_roles_expanded AS (
    SELECT er.fk_role_id
    FROM selected_employee se
    JOIN employee_roles er ON se.employee_id = er.fk_employee_id
),
training_status_base AS (
    SELECT 
        ap.procedure_name,
        MAX(
            CASE 
                WHEN t.ticket_id IS NOT NULL THEN 3
                WHEN rtt.required_type = 1 THEN 2
                WHEN rtt.required_type = 2 THEN 1
                ELSE 0
            END
        ) AS StatusRank
    FROM active_procedures ap
    CROSS JOIN selected_employee se
    LEFT JOIN employee_roles er ON er.fk_employee_id = se.employee_id
    LEFT JOIN required_training_types rtt 
        ON rtt.fk_role_id = er.fk_role_id 
        AND rtt.fk_treat_procedure_id = ap.procedure_id
    LEFT JOIN tickets t 
        ON t.fk_employee_id = se.employee_id 
        AND t.fk_treat_procedure_id = ap.procedure_id
    GROUP BY ap.procedure_name
)
SELECT 
    procedure_name,
    CASE StatusRank
        WHEN 3 THEN 'Training Completed'
        WHEN 2 THEN 'Training Missing'
        WHEN 1 THEN 'Training Optional'
        ELSE 'Place holder'
    END AS TrainingStatus
FROM training_status_base
ORDER BY procedure_name;
)

DATABASE SCRIPT
(
DROP TABLE IF EXISTS required_training_types;
DROP TABLE IF EXISTS tickets;
DROP TABLE IF EXISTS employee_roles;
DROP TABLE IF EXISTS employees;
DROP TABLE IF EXISTS treat_procedures;
DROP TABLE IF EXISTS revisions;
DROP TABLE IF EXISTS treat_roles;

CREATE TABLE treat_roles (
    role_id INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
    role_name VARCHAR(50) NOT NULL
);

CREATE TABLE employees (
    employee_id INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
    employee_email VARCHAR(100) NOT NULL,
    employee_initials VARCHAR(25) NOT NULL,
    employee_is_active BIT NOT NULL
);

CREATE TABLE employee_roles (
    fk_employee_id INT NOT NULL FOREIGN KEY REFERENCES employees (employee_id),
    fk_role_id INT NOT NULL FOREIGN KEY REFERENCES treat_roles (role_id),
    PRIMARY KEY (fk_employee_id, fk_role_id)
);

CREATE TABLE revisions (
    revision_id INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
    revision INT NOT NULL,
    revision_is_active BIT NOT NULL,
    revision_history_text VARCHAR(MAX) NOT NULL
);

CREATE TABLE treat_procedures (
    procedure_id INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
    procedure_name VARCHAR(50) NOT NULL,
    fk_revision_id INT NOT NULL FOREIGN KEY REFERENCES revisions (revision_id)
);

CREATE TABLE tickets (
    ticket_id INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
    submission_date DATE NOT NULL,
    fk_employee_id INT NOT NULL FOREIGN KEY REFERENCES employees (employee_id),
    fk_treat_procedure_id INT NOT NULL FOREIGN KEY REFERENCES treat_procedures (procedure_id),
    training_types VARCHAR(50) NOT NULL
);

CREATE TABLE required_training_types (
    required_type INT NOT NULL,
    fk_role_id INT NOT NULL FOREIGN KEY REFERENCES treat_roles(role_id),
    fk_treat_procedure_id INT NOT NULL FOREIGN KEY REFERENCES treat_procedures (procedure_id),
    PRIMARY KEY (required_type, fk_role_id, fk_treat_procedure_id)
);
